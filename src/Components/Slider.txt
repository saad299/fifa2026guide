import React, { useState, useEffect, useRef } from "react";
import { motion, AnimatePresence } from "framer-motion";

// Import your images (adjust names to your actual files)
import img1 from "../assets/img1.jpg";
import img2 from "../assets/img2.jpg";
import img3 from "../assets/img3.jpg";

const slides = [
  { id: 1, src: img1, alt: "Slide 1" },
  { id: 2, src: img2, alt: "Slide 2" },
  { id: 3, src: img3, alt: "Slide 3" },
];

export default function KenBurnsSlider({
  className = "",
  slideDuration = 6000, // ms each slide stays visible
}) {
  const [index, setIndex] = useState(0);
  const timeoutRef = useRef(null);

  useEffect(() => {
    // set up interval-like behavior with timeout so we can control lifecycle cleanly
    const start = () => {
      timeoutRef.current = setTimeout(() => {
        setIndex((prev) => (prev + 1) % slides.length);
      }, slideDuration);
    };

    start();
    return () => {
      if (timeoutRef.current) {
        clearTimeout(timeoutRef.current);
      }
    };
  }, [index, slideDuration]);

  // motion variants for fade and ken burns (slow zoom-in)
  const imageVariants = {
    enter: (custom) => ({
      opacity: 0,
      scale: 1,
    }),
    center: (custom) => ({
      opacity: 1,
      scale: 1.08, // Ken Burns target scale (slow zoom-in)
      transition: {
        duration: slideDuration / 1000, // animate zoom across the whole slide duration
        ease: "linear",
      },
    }),
    exit: (custom) => ({
      opacity: 0,
      scale: 1.08,
      transition: { duration: 0.8, ease: "easeOut" },
    }),
  };

  return (
    <div
      className={`relative w-full overflow-hidden ${className}`}
      aria-roledescription="carousel"
    >
      {/* Slides */}
      <div className="relative w-full h-[500px] md:h-[600px]">
        <AnimatePresence initial={false} mode="wait">
          {slides.map((slide, i) =>
            i === index ? (
              <motion.div
                key={slide.id}
                className="absolute inset-0"
                custom={i}
                initial="enter"
                animate="center"
                exit="exit"
                variants={imageVariants}
              >
                {/* Background image cover */}
                <div
                  className="absolute inset-0 bg-center bg-cover"
                  style={{ backgroundImage: `url(${slide.src})` }}
                  aria-hidden="true"
                />

                {/* If you prefer an <img> tag for SEO or LCP, uncomment below and comment the background div above */}
                {/* <motion.img
                  src={slide.src}
                  alt={slide.alt}
                  className="w-full h-full object-cover"
                  initial="enter"
                  animate="center"
                  exit="exit"
                  variants={imageVariants}
                /> */}

                {/* Optional overlay content (title/button) on top of the image */}
                <div className="absolute inset-0 flex items-end md:items-center justify-start p-6 md:p-12">
                  <div className="max-w-xl text-white">
                    <h2 className="text-2xl md:text-4xl font-bold drop-shadow-lg">
                      {/* Example dynamic title — replace with your own */}
                      {slide.alt}
                    </h2>
                    <p className="mt-2 hidden md:block drop-shadow">
                      A subtle subtitle that sits on top of the slide.
                    </p>
                  </div>
                </div>
              </motion.div>
            ) : null
          )}
        </AnimatePresence>
      </div>

      {/* Dots / indicators */}
      <div className="absolute bottom-4 left-1/2 transform -translate-x-1/2 flex gap-2 z-20">
        {slides.map((_, i) => (
          <button
            key={i}
            onClick={() => setIndex(i)}
            aria-label={`Go to slide ${i + 1}`}
            className={`w-3 h-3 rounded-full transition-all ${
              i === index ? "scale-110 bg-white" : "bg-white/50"
            }`}
          />
        ))}
      </div>
    </div>
  );
}
